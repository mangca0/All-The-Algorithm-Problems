==**及时移除无用数据，保证队列/栈的有序性**==
单调队列=单调栈+滑动窗口

高级语言用双端队列维护，c语言如下
```c
int fun (int *a,int n,int k)
{
	// front为队首指针，rear为队尾的后一位指针
	// 当front==rear时，队列为空
    int ans = 0, l = 0;
    int maxQueue[n + 10], minQueue[n + 10];
    int maxFront = 0, maxRear = 0, minFront = 0, minRear = 0;
    for(int i = 0; i < n; i ++)
    {
	    // 入
        int x = a[i];
        while(maxFront != maxRear && a[maxQueue[maxRear - 1]] <= x)
            maxRear --;
        maxQueue[maxRear ++] = i;
        while(minFront != minRear && a[minQueue[minRear - 1]] >= x)
            minRear --;
        minQueue[minRear ++] = i;

		// 出
        while(a[maxQueue[maxFront]] - a[minQueue[minFront]] > k)
        {
            if(maxQueue[maxFront] == l)maxFront ++;
            if(minQueue[minFront] == l)minFront ++;
            l ++;
        }

		// 更新答案
        ans = (i - l + 1 > ans ? i - l + 1 : ans);
    }
    return ans;
}
```
