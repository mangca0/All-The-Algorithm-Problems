struct LNode 
{ 
	ElementType Data[MAXSIZE];
	Position Last; /* 保存线性表中最后一个元素的位置 */ 
};

List MakeEmpty()
{
    List L = (List)malloc(sizeof(struct LNode));
    L -> Last = -1;
    return L;
}

Position Find( List L, ElementType X )
{
    int n = L -> Last;
    for(int i = 0; i <= n; i ++)
        if(L -> Data[i] == X)return i;
    return ERROR;
}

bool Insert( List L, ElementType X, Position P )
{
    int n = L -> Last;
    if(n == MAXSIZE - 1)
    {
        printf("FULL");
        return false;
    }
    if(P < 0 || P > n + 1)
    {
        printf("ILLEGAL POSITION");
        return false;
    }
    for(int i = n; i >= P; i --)L -> Data[i + 1] = L -> Data[i];
    L -> Data[P] = X;
    L -> Last ++;
    return true;
}

bool Delete( List L, Position P )
{
    int n = L -> Last;
    if(P < 0 || P > n)
    {
        printf("POSITION %d EMPTY", P);
        return false;
    }
    for(int i = P + 1; i <= n; i ++)L -> Data[i - 1] = L -> Data[i];
    L -> Last --;
    return true;
}
