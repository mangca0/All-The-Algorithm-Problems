维护相邻元素

操作：删除，插入前，插入后

```cpp
int tail;
int e[N], p[N];

inline void del(int x)
{
  if(e[x] == -1)e[p[x]] = -1, tail = p[x];
  else
  {
    e[p[x]] = e[x];
    p[e[x]] = p[x];
  }
}

inline void ins_front(int x, int y)
{
  e[x] = y;
  p[x] = p[y];
  e[p[y]] = x;
  p[y] = x;
}

inline void ins_back(int x, int y)
{
  if(e[y] == -1)
  {
    p[x] = y;
    e[y] = x;
    e[x] = -1;
    tail = x;
  }
  else
  {
    p[x] = y;
    e[x] = e[y];
    e[y] = x;
    p[e[x]] = x;
  }
}

for(int i = 1; i <= n; i ++)
{
  e[i] = i + 1;
  p[i] = i - 1;
}
e[0] = 1, e[n] = -1, tail = n;
