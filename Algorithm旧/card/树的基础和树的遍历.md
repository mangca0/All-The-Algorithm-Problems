inline void dfs1(int x)  
{  
    cout << x << ' ';  
    if(ls[x])dfs1(ls[x]);  
    if(rs[x])dfs1(rs[x]);  
}  
  
//中序遍历  
inline void dfs2(int x)  
{  
    if(ls[x])dfs2(ls[x]);  
    cout << x << ' ';  
    if(rs[x])dfs2(rs[x]);  
}  
  
//后序遍历  
inline void dfs3(int x)  
{  
    if(ls[x])dfs3(ls[x]);  
    if(rs[x])dfs3(rs[x]);  
    cout << x << ' ';  
}  
  
inline void bfs()  
{  
    queue<int> q;  
    q.push(1);  
    while(q.size())  
    {  
        int x = q.front();q.pop();  
        cout << x << ' ';  
        if(ls[x])q.push(ls[x]);  
        if(rs[x])q.push(rs[x]);  
    }  
}  
  
inline void init()  
{  
    int n;cin >> n;  
    for(int i = 1; i <= n; i ++)cin >> ls[i] >> rs[i];  
    dfs1(1);  
    cout << endl;  
    dfs2(1);  
    cout << endl;  
    dfs3(1);  
    cout << endl;  
    bfs();  
}
