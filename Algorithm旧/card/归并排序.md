其实是 **”排序“+”归并“**
```cpp
inline void MergeSort(int a[], int l, int r)
{
    if(l == r)return;
	
	// “排序”
    int mid = l + r >> 1;
    MergeSort(a, l, mid);
    MergeSort(a, mid + 1, r);
    
    // “归并
    int b[100] = {0};
    int pl = l, pr = mid + 1, pb = l;
    while(pl <= mid || pr <= r)
    {
        if(pl > mid)
            b[pb ++] = a[pr ++];
        else if(pr > r)
            b[pb ++] = a[pl ++];
        else
        {
            if(a[pl] <= a[pr])
                b[pb ++] = a[pl ++];
            else
                b[pb ++] = a[pr ++];
        }
    }

    for(int i = l; i <= r; i ++)a[i] = b[i];
}
```
