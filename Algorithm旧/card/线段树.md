```cpp
#define lson (root << 1)
#define rson (root << 1 | 1)

const int N = 2e5 + 9;
int a[N], t[N << 2], lazy[N << 2];

inline void build(int root, int l, int r)
{
	//停（完全覆盖）
    if(l == r)
    {
        t[root] = a[l];
        return;
    }
    int mid = l + r >> 1;
    //分
    build(lson, l, mid);
    build(rson, mid + 1, r);
    //合
    t[root] = t[lson] + t[rson];
}

inline void push_down(int root, int l, int r)
{
    if(!lazy[root])return;
    int mid = l + r >> 1, val = lazy[root];
    lazy[lson] += val, lazy[rson] += val;
    t[lson] += val * (mid - l + 1), t[rson] += val * (r - mid);
    lazy[root] = 0;
}

inline int query(int L, int R, int root, int l, int r)
{
	//停（完全覆盖）
    if(L <= l && r <= R)return t[root];
    push_down(root, l, r);
    int mid = l + r >> 1, ans = 0;
    //分
    if(L <= mid)ans += query(L, R, lson, l, mid);
    if(R > mid)ans += query(L, R, rson, mid + 1, r);
    //合
    return ans;
}

inline void updata(int L, int R, int root, int l, int r, int val)
{
	//停（完全覆盖）
    if(L <= l && r <= R)
    {
        lazy[root] += val;
        t[root] += val * (r - l + 1);
        return;
    }
    push_down(root, l, r);
    int mid = l + r >> 1;
    //分
    if(L <= mid)updata(L, R, lson, l, mid, val);
    if(R > mid)updata(L, R, rson, mid + 1, r, val);
    //合
    t[root] = t[lson] + t[rson];
}
```
