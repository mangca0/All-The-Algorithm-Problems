用于处理，当存在特殊起点集合，求集合内任意两点的最短距离的问题。

引入次短路是为了规避多源同时扩散时，产生的自环的假路径。

**相同：**

都要在松弛过程中记录路径**源点**。

都要规定最短路和次短路源点不同。

**不同：**

对比 Dijkstra，BFS 在处理平权图时，先到的点一定更近，所以就不存在 Dijkstra 里那种「用更短距离更新」、「用最短距离更新次短距离」的操作。

BFS 在第一次找到相同色块时，只要源点不同，就是最短距离，可以直接中止程序。

Dijkstra
```java
import java.io.*;
import java.util.*;

public class Main {
    static final long INF = Long.MAX_VALUE / 2;

    public static void main(String[] args) throws IOException {
        FastReader in = new FastReader();
        String tStr = in.next();
        if (tStr == null) return;
        int T = Integer.parseInt(tStr);

        while (T-- > 0) {
            int n = in.nextInt();
            int m = in.nextInt();

            List<int[]>[] g = new ArrayList[n + 1];
            for (int i = 1; i <= n; i++) {
                g[i] = new ArrayList<>();
            }

            for (int i = 0; i < m; i++) {
                int u = in.nextInt();
                int v = in.nextInt();
                int w = in.nextInt();
                g[u].add(new int[]{v, w});
                g[v].add(new int[]{u, w}); 
            }

            int k = in.nextInt();

            long[] dist1 = new long[n + 1];
            int[] orig1 = new int[n + 1];
            long[] dist2 = new long[n + 1];
            int[] orig2 = new int[n + 1];

            Arrays.fill(dist1, INF);
            Arrays.fill(dist2, INF);

            PriorityQueue<long[]> pq = new PriorityQueue<>((a, b) -> Long.compare(a[0], b[0]));

            for (int i = 0; i < k; i++) {
                int x = in.nextInt();
                dist1[x] = 0;
                orig1[x] = x;
                pq.add(new long[]{0, x, x});
            }

            while (!pq.isEmpty()) {
                long[] curr = pq.poll();
                long d = curr[0];
                int u = (int) curr[1];
                int origin = (int) curr[2];

                if (d > dist2[u]) continue;

                for (int[] edge : g[u]) {
                    int v = edge[0];
                    long w = edge[1];
                    long nxtD = d + w;

                    if (nxtD < dist1[v]) {
                        if (origin != orig1[v]) {
                            dist2[v] = dist1[v];
                            orig2[v] = orig1[v];
                            pq.add(new long[]{dist2[v], v, orig2[v]});
                        }
                        dist1[v] = nxtD;
                        orig1[v] = origin;
                        pq.add(new long[]{nxtD, v, origin});
                    } else if (origin != orig1[v] && nxtD < dist2[v]) {
                        dist2[v] = nxtD;
                        orig2[v] = origin;
                        pq.add(new long[]{nxtD, v, origin});
                    }
                }
            }

            long minAns = INF;
            for (int i = 1; i <= n; i++) {
                if (orig1[i] != 0 && orig2[i] != 0 && orig1[i] != orig2[i]) {
                    minAns = Math.min(minAns, dist1[i] + dist2[i]);
                }
            }
            System.out.println(minAns);
        }
    }

    static class FastReader {
        BufferedReader br;
        StringTokenizer st;

        public FastReader() {
            br = new BufferedReader(new InputStreamReader(System.in));
        }

        String next() {
            while (st == null || !st.hasMoreElements()) {
                try {
                    String line = br.readLine();
                    if (line == null) return null;
                    st = new StringTokenizer(line);
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            return st.nextToken();
        }

        int nextInt() {
            return Integer.parseInt(next());
        }
    }
}
```

BFS
```java
import java.io.*;
import java.util.*;

public class Main {
    static final int INF = (int) 1e9;

    public static void main(String[] args) throws IOException {
        FastReader in = new FastReader();
        String tStr = in.next();
        if (tStr == null) return;
        int T = Integer.parseInt(tStr);

        while (T-- > 0) {
            int n = in.nextInt();
            int m = in.nextInt();
            int k = in.nextInt();

            int[] color = new int[n + 1];
            for (int i = 1; i <= n; i++) {
                color[i] = in.nextInt();
            }

            List<Integer>[] g = new ArrayList[n + 1];
            for (int i = 1; i <= n; i++) g[i] = new ArrayList<>();
            for (int i = 0; i < m; i++) {
                int u = in.nextInt();
                int v = in.nextInt();
                g[u].add(v);
            }

            int[] dist1 = new int[n + 1];
            int[] orig1 = new int[n + 1];
            int[] dist2 = new int[n + 1];
            int[] orig2 = new int[n + 1];

            int maxQ = n * 2 + 10;
            int[] qD = new int[maxQ];
            int[] qU = new int[maxQ];
            int[] qOrig = new int[maxQ];

            boolean profChenCorrect = true;

            int limitD = Math.min(k - 1, 50);

            for (int c = 1; c <= 50; c++) {
                if (!profChenCorrect) break;

                for (int i = 1; i <= n; i++) {
                    dist1[i] = INF; dist2[i] = INF;
                    orig1[i] = 0; orig2[i] = 0;
                }

                int head = 0, tail = 0;
                int colorCount = 0;

                for (int i = 1; i <= n; i++) {
                    if (color[i] == c) {
                        dist1[i] = 0;
                        orig1[i] = i;
                        qD[tail] = 0;
                        qU[tail] = i;
                        qOrig[tail] = i;
                        tail++;
                        colorCount++;
                    }
                }

                if (colorCount < 2) continue;

                int shortestEdgePath = INF;

                while (head < tail) {
                    int d = qD[head];
                    int u = qU[head];
                    int origin = qOrig[head];
                    head++;

                    if (d >= limitD) break;

                    for (int v : g[u]) {
                        int nxtD = d + 1;

                        if (color[v] == c && origin != v) {
                            shortestEdgePath = nxtD;
                            break;
                        }

                        if (dist1[v] == INF) {
                            dist1[v] = nxtD;
                            orig1[v] = origin;
                            qD[tail] = nxtD; qU[tail] = v; qOrig[tail] = origin; tail++;
                        }
                        else if (origin != orig1[v] && dist2[v] == INF) {
                            dist2[v] = nxtD;
                            orig2[v] = origin;
                            qD[tail] = nxtD; qU[tail] = v; qOrig[tail] = origin; tail++;
                        }
                    }

                    if (shortestEdgePath != INF) break;
                }

                if (shortestEdgePath + 1 <= k) {
                    profChenCorrect = false;
                }
            }

            System.out.println(profChenCorrect ? "YES" : "NO");
        }
    }

    static class FastReader {
        BufferedReader br;
        StringTokenizer st;
        public FastReader() {
            br = new BufferedReader(new InputStreamReader(System.in));
        }
        String next() {
            while (st == null || !st.hasMoreElements()) {
                try {
                    String line = br.readLine();
                    if (line == null) return null;
                    st = new StringTokenizer(line);
                } catch (IOException e) { e.printStackTrace(); }
            }
            return st.nextToken();
        }
        int nextInt() { return Integer.parseInt(next()); }
    }
}
```
