	- ###### 二进制枚举 + 模拟   (可优化为上图限制条件)
	```
	#include<bits/stdc++.h>  
	#define int long long  
	#define endl "\n"  
	using namespace std;  
	  
	const int N = 10;  
	int a[N], b[N];  
	int ans = 7;  
	vector<int> c[6] = {{1, 5, 7}, {1, 2, 5, 8}, {1, 3, 5}, {2, 4, 8}, {3, 6}, {4, 6, 8}};  
	  
	inline void init()  
	{  
	  for (int i = 0; i <= 63; i++)  
	  {  
	    int flag = 1;  
	    int cnt = 0;  
	    memset(a, 0, sizeof(a));  
	    memset(b, 0, sizeof(b));  
	    for (int j = 0; j <= 5; j++)  
	      {  
	        if (i >> j & 1)a[5 - j]++;  
	      }  
	    for (int j = 0; j <= 5; j++)  
	    {  
	      if (a[j])  
	      {  
	        for (auto k : c[j])b[k - 1]++;  
	      }  
	    }  
	    flag = 1;  
	    for (int j = 0; j <= 7; j++)  
	    {  
	      if (!b[j])  
	      {  
	        flag = 0;  
	        break;  
	      }  
	      if (a[j])cnt++;  
	    }  
	    if (!flag)continue;  
	    ans = min(ans, cnt);  
	  }  
	  cout << ans << endl;  
	}  
	  
	signed main()  
	{  
	  ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);  
	  init();  
	  return 0;  
	}
	```
