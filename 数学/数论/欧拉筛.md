**枚举倍数i，而不是质因子**
```
int cnt = 0;//统计有多少个质数
memset(vis, 0, sizeof(vis));//vis值为0为质数，为1为合数
for(int i = 2; i <= n; i ++)
{
	if(!vis[i])prime[++ cnt] = i;//prime存储所有质数 
	for(int j = 1; j <= cnt; j ++)
	{
		if(i * prime[j] > n)break;//超出n的范围
		vis[i * prime[j]] = 1;
		if(i % prime[j] == 0)break;//说明prime[j]已经不是i * prime[j]的最小质因数了
	}
}
```
